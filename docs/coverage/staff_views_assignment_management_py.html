39 class used to display each row in the table

46 self.job = assignment.job 

47 self.route_number = assignment.job.get_route_number()

55 def __lt__(self, other_row): 

56 """ 

57 used with sorted is called 

58 this allows the route to all get displayed first by route number 

59 then all other jobs by alphabetical 

60 """ 

61 if (self.route_number != -1 and other_row.route_number != -1) and ( 

62 self.route_number != other_row.route_number 

63 ): 

64 # both are routes but not the same route 

65 return self.route_number < other_row.route_number 

66 elif self.route_number == -1 and other_row.route_number != -1: 

67 # one is not a route 

68 # routes will get priority in default sort 

69 return False 

70 elif self.route_number != -1 and other_row.route_number == -1: 

71 # one is not a route 

72 # routes will get priority in default sort 

73 return True 

74 elif self.job.job_type.name != other_row.job.job_type.name: 

75 # this will allow for the packers and shuttles to be separated 

76 return self.job.job_type.name < other_row.job.job_type.name 

77 elif self.job.name != other_row.job.name: 

78 # they are not the same job 

79 return self.job.name < other_row.job.name 

80 else: 

81 if self.volunteer and other_row.volunteer: 

82 self_name = f"{self.volunteer.user.first_name.lower()} {self.volunteer.user.last_name.lower()}" 

83 other_row_name = f"{other_row.volunteer.user.first_name.lower()} {other_row.volunteer.user.last_name.lower()}" 

84 return self_name < other_row_name 

85 elif self.volunteer: 

86 # one will display as open route 

87 self_name = f"{self.volunteer.user.first_name.lower()} {self.volunteer.user.last_name.lower()}" 

88 return self_name < OPEN_ROUTE.lower() 

89 else: 

90 # other_row.volunteer is not None 

91 # one will display as open route 

92 other_row_name = f"{other_row.volunteer.user.first_name.lower()} {other_row.volunteer.user.last_name.lower()}"

121def generate_assignments_display(single_assignment_list): 

122 """ 

123 :param single_assignment_list: list of SingleDayAssignment objects 

124 takes in a list of SingleDayAssignments and condenses 

125 down to MutipleDayAssignmentDisplays where possible, then returns 

126 comma joined list of MultipleDayAssignments and SingleDayAssignments as strings 

127 """ 

128 # lists to store single day and multiple day assignments 

129 single_days = [] 

130 multiple_days = [] 

131 # hidden assignments list for indexing in the filter 

132 # this way a query for '1st Mon' will still match a row that says 'Mondays' 

133 hidden_days = [] 

134 # loop over days of the week 

135 for i in range(1, 8): 

136 matching_assignments = list(filter(lambda item: item.day_of_week == i, single_assignment_list)) 

137 if len(matching_assignments) == 5: 

138 # assignment happens weekly 

139 multiple_days.append(MultipleDayAssignment(i)) 

140 hidden_days += matching_assignments 

141 else: 

142 # can't condense 

143 single_days += matching_assignments 

144 

145 # sort based on day of week 

146 multiple_days.sort(key=lambda item: item.day_of_week) 

147 # sort based on week and then day 

148 single_days.sort(key=lambda item: (item.week_of_month, item.day_of_week)) 

149 

150 # return visible assignments 

151 return ', '.join([str(item) for item in multiple_days] + [str(item) for item in single_days])

155def manage_assignments_table(request, job_pk=None, vol_pk=None): 

156 """ 

157 generates the manage assignments table, 

158 this will get displayed in the manage_assignments template 

159 it is separated because it runs slowly 

160 job_pk will be 0 by when vol_pk is passed 

161 """ 

162 row_dict = {} 

163 # get the appropriate assignments 

164 if job_pk is not None and job_pk != 0: 

165 query = Assignment.objects.filter(job=get_object_or_404(Job, pk=job_pk)) 

166 elif vol_pk is not None and job_pk == 0: 

167 query = Assignment.objects.filter(volunteer=get_object_or_404(Volunteer, pk=vol_pk)) 

168 else: 

169 query = Assignment.objects.all() 

170 

171 for item in query: 

172 # create and edit the row items 

173 if (item.volunteer, item.job) in row_dict: 

174 row = row_dict[(item.volunteer, item.job)] 

175 row.hidden_assignments.append( 

176 SingleDayAssignment(item.week_of_month, item.day_of_week) 

177 ) 

178 else: 

179 row_dict[(item.volunteer, item.job)] = Row(item) 

180 

181 # create the visible and hidden assignments 

182 rows = row_dict.values() 

183 for row in rows: 

184 row.visible_assignments = generate_assignments_display(row.hidden_assignments) 

185 

186 return render(request, "manage-assignments-table.html", {"rows": sorted(rows), "open_job": OPEN_ROUTE})
