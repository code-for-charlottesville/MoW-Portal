16def query_substitutions_table(number_of_days, vol): 

17 """ 

18 :param number_of_days: number of days in advance 

19 :return: Substitution queryset 

20 queries the substitutions table for the substitutions 

21 from today's date until however many days are passed in 

22 """ 

23 startdate = datetime.date.today() 

24 # Change the timedelta to look for the number of weeks in advance 

25 enddate = startdate + datetime.timedelta(days=number_of_days - 1) 

26 sub_query = Substitution.objects.filter(date__range=[startdate, enddate]) 

27 sub_day_dict = {} # will be date string:subs for that day. 

28 for i in range(number_of_days): 

29 search_day = datetime.date.today() + datetime.timedelta(days=i) 

30 # lets skip weekends for now 

31 day_of_week = search_day.isoweekday() 

32 if day_of_week == 6 or day_of_week == 7: 

33 continue 

34 sub_day_query = sub_query.filter(date=search_day, volunteer=vol) 

35 sub_day_dict[search_day.strftime("%A, %B %d")] = sub_day_query 

36 

37 return sub_day_dict
